Promises in JavaScript

What is a Promise?

The Promise object represents the eventual completion (or failure) of an asynchronous operation and its resulting value.

Create Promises in JavaScript


					// Create a Promise
					const myPromise = new Promise((resolve, reject) => {
							const randomNumber = Math.random(); // Generate a random number
					
							if (randomNumber > 0.5) {
// If the number is greater than 0.5, the promise is fulfilled
									resolve(randomNumber); // Pass the random number as the result
							} else {
// If the number is less than or equal to 0.5, the promise is rejected
									reject(new Error("The random number is too low!"));
							}
					});

Promise Methods

then()

finally()


							fetch('https://api.example.com/data')
							.then(response => response.json())
							.then(data => {
								// Work with the fetched data here
								console.log('Data received:', data);
							})
							.catch(error => {
								// Handle any errors during the API call
								console.error('Error fetching data:', error);
							})
							.finally(() => {
								// Perform cleanup tasks here
								console.log('API call completed.');
							});

Promise.all()


         const promise1 = fetch('url1');
         const promise2 = fetch('url2');
         Promise.all([promise1, promise2])
           .then(results => {
        // results contains an array with the fulfilled values of both promises
            })
           .catch(error => {
        // Handle any errors from any of the promises
           });

Promise.race()


					const promise1 = fetch('url1');
					const promise2 = fetch('url2');
					Promise.race([promise1, promise2])
						.then(result => {
							// result is the value of the first promise to fulfill
						})
						.catch(error => {
							// Handle any errors from the first promise to reject
						});
